5   /**
6    * An object which eventually becomes immutable.
7    * <p/>
8    * It might be immutable from creation, or it might start as mutable and become immutable when
9    * {@code fixAsVersion(..)} is called.
10   * <p/>
11   * The convention to use is {@code X getX()} for accessors and
12   * {@code EventuallyImmutable withX(newX)} for mutators.
13   * <p/>
14   * Whilst {@code this.isMutable()}, mutators mutate the internal data and return the same object.
15   * Once immutable, mutators create a mutable clone and mutations are applied to it, i.e.
16   * {@code createMutableClone().withX(newX)}.
17   * <p/>
18   * Mutations can therefore be strung together, e.g. {@code obj.withX(newX).withY(newY).withZ(newZ)}
19   * will return {@code obj} with the mutations if {@code obj.isMutable()} or
20   * {@code obj.createMutableClone()} with the mutations otherwise.
21   * <p/>
22   * Immutable objects are made aware of replacements ready to take their place with
23   * {@code original.replaceWith(replacement)}.  It then updates its {@link BestSoFar.framework.core.helper.VersionInfo}
24   * information and generates the {@code Version} for the replacement,
25   * which is passed in {@code replacement.fixAsVersion(version)} and ensures the replacement is
26   * finalised (i.e. immutable).
27   * <p/>
28   * An object may be finalised multiple times to update it with new version information.  No new
29   * objects are created in doing so, and the only effect is to save teh new version information.
30   *
31   * @see BestSoFar.framework.core.helper.MutabilityHelper MutabilityHelper
32   */
33  public interface EventuallyImmutable extends Deletable {
34      /**
35       * Create a mutable clone of this object.  It will remain mutable until passed as a
36       * {@code replacement} to {@code replaceWith(replacement)}, at which point
37       * {@code fixAsVersion(..)} will be called to effect the change to an immutable object.
38       *
39       * @return a mutable clone.
40       * @throws RuntimeException if this object is still mutable.
41       */
42      EventuallyImmutable createMutableClone();
43  
44      /**
45       * If this object is mutable, this method will make it immutable. In any case,
46       * it will save the new version information.
47       * <p/>
48       * Calling {@code replaceWith(replacement)} creates the new
49       * {@link BestSoFar.framework.core.helper.VersionInfo version} information and passes it to
50       * {@code replacement.beforeFixAsVersion(version)}.
51       *
52       * @param versionInfo the new version information for this object.
53       */
54      void fixAsVersion(VersionInfo versionInfo);
55  
56      /**
57       * Returns true iff this object is mutable, meaning it was created as mutable and has not
58       * been finalised.
59       *
60       * @return true iff this object is mutable.
61       */
62      boolean isMutable();
63  
64      /**
65       * Propose a replacement for this object.  How this is handled is determined by the concrete
66       * class implementing this.
67       * <p/>
68       * This must have no next version, and the replacement must have no previous version -
69       * otherwise a RuntimeException is thrown.
70       * <p/>
71       * As an example, a sequence of versions between {@code EventuallyImmutable}
72       * objects {@code start} and {@code end} can be contracted using
73       *
74       * <pre>{@code
75       * start.discardNext();  // start.getNext() == null
76       * end.discardPrevious(); // end.getPrevious() == null
77       * start.replaceWith(end);
78       * }</pre>
79       *
80       * @param replacement the replacement to propose.
81       * @throws RuntimeException if this object is still mutable,
82       *                          or has already been replaced or deleted.
83       */
84      void replaceWith(EventuallyImmutable replacement);
85  
86      /**
87       * Discards any replacement to this object and ensures this object is not deleted,
88       * therefore allowing for it to be replaced by another {@code EventuallyImmutable} object.
89       * In doing so, it makes the next version discard its previous version.
90       */
91      void discardNext();
92  
93      /**
94       * Discard older versions of this, so they may be freed by the garbage collector.  In doing
95       * so it makes the previous version discard its next version.
96       */
97      void discardPrevious();
98  
99      /**
100      * Gets the {@link BestSoFar.framework.core.helper.VersionInfo} object that describes this object.  It contains
101      * information about this version of the {@code EventuallyImmutable} object,
102      * and its earlier and later versions.
103      *
104      * @return the version information about this object.
105      */
106     VersionInfo versionInfo();
