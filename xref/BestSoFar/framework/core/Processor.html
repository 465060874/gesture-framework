12  /**
13   * The top-level interface for all parts of the workflow framework.
14   * <p/>
15   * A {@code Processor} can process input data (inside a {@link Mediator}) and return output data
16   * inside another {@code Mediator}.
17   * <p/>
18   * Before being asked to process a {@code Mediator}, it will first process a training batch.  A
19   * typical sequence that the {@code Processor} methods are called in is:
20   *   <p/>
21   *   - {@code notify(List<Mediator<O>>)} : from observed Processors as they process the
22   *       training batch.  Features of the supplied data can be gathered here. NB. this is only
23   *       called if the {@code Processor} is a {@link ElementObserver}.
24   *   <p/>
25   *   - {@code processTrainingBatch(..)} : when it's this object's turn to process the
26   *     training batch.
27   *   <p/>
28   *   - {@code processCompletedTrainingBatch(..)} : when the completed training batch
29   *     is travelling backward over the workflow that created it. This is where the
30   *     {@code Processor} may learn from the training batch (possibly correlated against
31   *     previously-gathered data features).
32   * <p/>
33   * This sequence allows allows for the {@code Processor} to prepare itself,
34   * for example it might choose to process the data in a certain way that worked well during
35   * training.
36   */
37  public interface Processor<I,O> extends EventuallyImmutable {

44      boolean isValid();

47       * Process the input {@link Mediator} object and return the result in an output {@code
48       * Mediator} object.
49       *
50       * @param input input data in a {@code Mediator} object.
51       * @return result of processing the input data, inside a {@code Mediator<O>} object.
52       * @throws ClassCastException if input cannot be cast to {@code Mediator<I>}.
53       */
54      Mediator<O> process(Mediator<?> input);

56      /**
57       * Given a list of input {@link Mediator} objects, produce all output {@code Mediator}
58       * objects that this {@link Processor} could possibly produce.  The input data will be for
59       * training, and if this {@code Processor} expects to be notified of prior {@code Processor}
60       * objects' completions, it will be notified by the {@code notify(List<Mediator<?>>)} method,
61       * which contains all training data.
62       * <p/>
63       * For elemental {@code Processors}, this means applying {@code process(input)} on each
64       * input, and for {@code Processors} comprising multiple {@link Workflow} objects this means
65       * processing each input using all appropriate {@code Workflows}.
66       *
67       * @param inputs list of {@code Mediator} objects containing input data.
68       * @return all output {@code Mediators} that this could possibly produce.
69       * @throws ClassCastException if any input cannot be cast to {@code Mediator<I>}.
70       */
71      List<Mediator<O>> processTrainingBatch(List<Mediator<?>> inputs);

80      /**
81       * This method is called after a training batch has been processed to completion, and returns
82       * a rolled-back version of the supplied {@link CompletedTrainingBatch}.
83       * <p/>
84       * The supplied {@code completedTrainingBatch} object contains the {@code Mediator} objects
85       * returned from the last call to {@code processTrainingBatch(..)}, and which of those went on
86       * to be successful.
87       * <p/>
88       * "Rolling-back" means reverting these output {@code Mediator<O>} objects to input
89       * {@code Mediator<I>} objects, and correctly marking which of those went on to be successful.
90       * <p/>
91       * {@code Processor} objects containing multiple ways of processing the same input will have
92       * created multiple outputs per input, in which case this method must choose which output
93       * would have been created had the input been given to {@code process(input)}.  Only if this
94       * output was successful will its corresponding input mediator will be marked as successful.
95       * <p/>
96       * If this {@code Processor} needs training or optimising, this is the method to do it.
97       * <p/>
98       * If this {@code Processor} intends to train a classifier to be used in {Code process(input)}
99       * to select the best strategy for each individual input data, it is advisable to create a
100      * classifier for each input {@code Mediator} object's {@link History} object.  This means
101      * that data that is created differently is classified differently.
102      * <p/>
103      * One approach would be to use a {@code Map<History, Classifier>} object that is queried in
104      * the {@code process(input)} method to get the appropriate
105      * {@link BestSoFar.framework.concretions.helper.Classifier} for the given
106      * {@code input.getHistory()} object.
107      * <p/>
108      * It is left to the concrete class to decide whether to do this or not (eg. the training set
109      * per classifier would be larger by not doing this).
110      *
111      * @param completedTrainingBatch the completed training batch, containing the {@link Mediator}
112      *                               objects returned from the last call to
113      *                               {@code processTrainingBatch(..)}, and which of those went on
114      *                               to be successful.
115      * @return the rolled-back {@code CompletedTrainingBatch}.
116      */
117     CompletedTrainingBatch<I> processCompletedTrainingBatch(
118             CompletedTrainingBatch<?> completedTrainingBatch);
