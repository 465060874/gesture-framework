12  /**
13   * A helper object that manages a child's parent in an {@link EventuallyImmutable} parent-child
14   * hierarchy.
15   */
16  public class ParentManager<C extends ChildOf<P> & EventuallyImmutable,
17                             P extends ParentOf<C> & EventuallyImmutable>
18          implements ChildOf<P> {
19  
20      @Getter private P parent;
21      private final C managedChild;
22  
23      /**
24       * Create a {@code ParentManager} to manage the given child's parent,
25       * which is initially set to null.
26       *
27       * @param managedChild the child who's parent this will manage.
28       */
29      public ParentManager(@NonNull C managedChild) {
30          this(managedChild, null);
31      }
32  
33      /**
34       * Create a {@code ParentManage} to manage the given child's given {@code parent}.
35       *
36       * @param managedChild the child who's parent this will manage.
37       * @param parent the child's parent.
38       */
39      public ParentManager(@NonNull C managedChild, P parent) {
40          this.managedChild = managedChild;
41          this.parent = parent;
42      }
43  
44      @Override
45      @SuppressWarnings("unchecked")
46      public C withParent(P newParent) {
47          if (getParent() == newParent)
48              return managedChild;
49  
50          if (managedChild.isMutable()) {
51              this.parent = newParent;
52              return managedChild;
53          }
54  
55          if (newParent != null)
56              newParent = (P) newParent.versionInfo().getLatest();
57  
58          C childClone = (C) managedChild.createMutableClone();
59          return (C) childClone.withParent(newParent);
60      }
61  
62      /**
63       * Called after the managed child has been deleted.
64       * <p/>
65       * If the child has a parent that wasn't deleted, this method removes the child from it.
66       */
67      @SuppressWarnings("unchecked")
68      public void afterDelete() {
69          if (getParent() != null && !getParent().isDeleted() && !getParent().isMutable()) {
70              List<C> siblings = new LinkedList<>(getParent().getChildren());
71              siblings.remove(managedChild);
72              P newParent = (P) getParent().createMutableClone();
73              newParent = (P) newParent.withChildren(siblings);
74              getParent().replaceWith(newParent);
75          }
76      }
77  
78      /**
79       * Called before the managed child is fixed.
80       * <p/>
81       * One of two scenarios caused this update
82       * <p/>
83       *     1. A property of the parent was changed
84       *     <p/>
85       *     2. A property of the managed child was changed
86       *
87       * @param versionInfo the new version information.
88       */
89      @SuppressWarnings("unchecked")
90      public void beforeFixAsVersion(VersionInfo versionInfo) {
91          // In scenario 1, the parent is updating its children to point to the new version of the
92          // parent.  There's nothing to do!
93  
94          // In scenario 2, the parent remains its old, immutable version.  We must create a new
95          // parent given the updated list of children.
96          if (getParent() != null && !getParent().isMutable() && managedChild.isMutable()) {
97              List<C> newSiblings = new LinkedList<>(getParent().getChildren());
98              VersionInfo.updateAllToLatest(newSiblings);
99              if (!newSiblings.contains(managedChild))
100                 newSiblings.add(managedChild);
101 
102             P oldParent = parent;
103             parent = (P) getParent().withChildren(newSiblings);
104             oldParent.replaceWith(parent);
105         }
106     }
