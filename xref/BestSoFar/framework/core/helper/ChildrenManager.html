13  /**
14   * A helper object that manages a list of children in an {@link EventuallyImmutable} parent-child
15   * hierarchy.
16   */
17  public class ChildrenManager<C extends ChildOf<P> & EventuallyImmutable,
18                               P extends ParentOf<C> & EventuallyImmutable>
19          implements ParentOf<C> {
20  
21      @Getter private List<C> children;
22      private final P managedParent;
23  
24      /**
25       * Constructs a {@code ChildrenManager} with an empty children list.
26       *
27       * @param managedParent the parent who's children this will manage.
28       */
29      public ChildrenManager(P managedParent) {
30          this(managedParent, Collections.<C>emptyList());
31      }
32  
33      /**
34       * Constructs a {@code ChildrenManager} with the given children list.  The list won't be
35       * mutated (i.e. it is fine for children to be {@code Collections.unmodifiableList}).
36       *
37       * @param managedParent the parent who's children this will manage.
38       * @param children
39       */
40      public ChildrenManager(@NonNull P managedParent, @NonNull List<C> children) {
41          this.managedParent = managedParent;
42          this.children = children;
43      }
44  
45      @Override
46      @SuppressWarnings("unchecked")
47      public P withChildren(@NonNull List<C> newChildren) {
48          if (managedParent.isMutable()) {
49              children = Collections.unmodifiableList(new LinkedList<>(newChildren));
50              return managedParent;
51          } else {
52              P parentClone = (P) managedParent.createMutableClone();
53              return (P) parentClone.withChildren(newChildren);
54          }
55      }
56  
57      /**
58       * Called after the managed parent has been deleted, to delete all its children..
59       */
60      public void afterDelete() {
61          for (C child : children)
62              child.delete();
63      }
64  
65      /**
66       * Called before the managed parent is being fixed.
67       * <p/>
68       * One of four scenarios caused this update:
69       * <p/>
70       *     1. A child has mutated
71       *     <p/>
72       *     2. Another property of the this parent has been changed
73       *     <p/>
74       *     3. A child has been deleted or moved to another parent
75       *     <p/>
76       *     4. A new child has been added to this parent
77       *
78       * @param versionInfo the new version information.
79       */
80      @SuppressWarnings("unchecked")
81      public void beforeFixAsVersion(VersionInfo versionInfo) {
82          if (!managedParent.isMutable())
83              return;
84  
85          List<C> latestChildren = new LinkedList<>(children);
86  
87          // Update all children to their latest versions, removing those that were deleted.
88          VersionInfo.updateAllToLatest(latestChildren);
89          children = new LinkedList<>();
90  
91          for (C child : latestChildren) {
92              if (child.getParent() == managedParent) {
93                  // The child has already been created pointing to managedParent as its
94                  // parent, so keep it as is.
95                  children.add(child);
96              } else {
97                  // The child needs to have managedParent set as its parent.
98                  C newChild = (C) child.withParent(managedParent);
99                  child.replaceWith(newChild);
100                 children.add(newChild);
101             }
102         }
103 
104         children = Collections.unmodifiableList(children);
105     }
106 
107     /**
108      * Called when the managed parent is having its replacement discarded,
109      * and discards all of its childrens' replacements.
110      */
111     public void discardNext() {
112         for (C child : children)
113             child.discardNext();
114     }
115 
116     /**
117      * Called when the managed parent is having its older versions discarded,
118      * and discards all of its childrens' older versions.
119      */
120     public void discardPrevious() {
121         for (C child : children)
122             child.discardPrevious();
123     }
