18  public class FingertipReducer extends AbstractElement {
19  
20      @Getter private int minFingerDepth, maxFingerAngle;
21  
22      public FingertipReducer() {
23          super(new TypeData(Fingertips.class, Fingertips.class));
24          minFingerDepth = 15;
25          maxFingerAngle = 150;
26      }
27  
28      public FingertipReducer(FingertipReducer oldElement) {
29          super(oldElement);
30          minFingerDepth = oldElement.getMinFingerDepth();
31          maxFingerAngle = oldElement.getMaxFingerAngle();
32      }
33  
34      @Override
35      public Mediator process(Mediator input) {
36          Fingertips allPoints = (Fingertips) input.getData();
37          List<CvPoint> newTips = new ArrayList<>();
38          List<CvPoint> newFolds = new ArrayList<>();
39          List<Float> newDepths = new ArrayList<>();
40  
41          int numPoints = allPoints.getTips().size();
42  
43          for (int i = 0; i < numPoints; ++i) {
44              if (allPoints.getDepths().get(i) < minFingerDepth)
45                  continue;
46  
47              // look at fold points on either side of a tip
48              int pdx = (i == 0) ? (numPoints - 1) : (i - 1); // predecessor of i
49              int sdx = (i == numPoints - 1) ? 0 : (i + 1);   // successor of i
50  
51              int angle = angleBetween(
52                      allPoints.getTips().get(i),
53                      allPoints.getFolds().get(pdx),
54                      allPoints.getFolds().get(sdx)
55              );
56  
57              if (angle >= maxFingerAngle)
58                  continue;      // angle between finger and folds too wide
59  
60              newTips.add(allPoints.getTips().get(i));
61              newFolds.add(allPoints.getFolds().get(i));
62              newDepths.add(allPoints.getDepths().get(i));
63          }
64  
65          return input.createNext(this, new Fingertips(newTips, newFolds, newDepths, allPoints.getSourceTaggedImage()));
66      }
67  
68      @Override
69      public Element createMutableClone() {
70          return new FingertipReducer(this);
71      }
72  
73      @Override
74      public boolean isValid() {
75          return true;
76      }

78      private int angleBetween(CvPoint tip, CvPoint next, CvPoint prev) {
79          return Math.abs((int) Math.round(
80                  Math.toDegrees(
81                          Math.atan2(next.x() - tip.x(), next.y() - tip.y()) -
82                                  Math.atan2(prev.x() - tip.x(), prev.y() - tip.y()))));
83      }
