27      @Override
28      public void view(Mediator mediator) {
29          Fingertips fingertips = (Fingertips) mediator.getData();
30          opencv_core.IplImage src = fingertips.getSourceTaggedImage().getImage();
31  
32          opencv_core.IplImage copiedImage = opencv_core.IplImage.create(cvGetSize(src), src.depth(), 3);
33          cvCvtColor(src, copiedImage, CV_GRAY2BGR);
34  
35          for (CvPoint tip : fingertips.getTips())
36              drawCircle(copiedImage, CvScalar.RED, tip);
37  
38          for (CvPoint fold : fingertips.getFolds())
39              drawCircle(copiedImage, CvScalar.GREEN, fold);
40  
41          super.view(copiedImage);
42      }
43  
44      private void drawCircle(IplImage image, CvScalar colour, CvPoint position) {
45          cvDrawCircle(image, position, 10, colour, 10, CV_AA, 0);
46      }

48      @Override
49      public String toString() {
50          return "Fingertip Viewer";
51      }
