31  
32          opencv_core.IplImage copiedImage = IplImage.create(cvGetSize(src), src.depth(), 3);
33          cvCvtColor(src, copiedImage, CV_GRAY2BGR);
34  
35          opencv_core.CvScalar colour = opencv_core.CvScalar.RED;
36          cvDrawContours(copiedImage, contours.getContour(), colour, colour, -1, 3, CV_AA);
37  
38          super.view(copiedImage);
39      }
