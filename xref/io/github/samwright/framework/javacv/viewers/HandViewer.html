26      @Override
27      public void view(Mediator mediator) {
28          Hand hand = (Hand) mediator.getData();
29          opencv_core.IplImage src = hand.getSourceTaggedImage().getImage();
30  
31          opencv_core.IplImage copiedImage = opencv_core.IplImage.create(cvGetSize(src), src.depth(), 3);
32          cvCvtColor(src, copiedImage, CV_GRAY2BGR);
33  
34          CvPoint centre = hand.getPalm().getCog();
35  
36          System.out.println("\n== Fingers:");
37  
38          for (int i = 0; i < hand.getDigitAngles().size(); ++i) {
39              double angle = hand.getDigitAngles().get(i);
40              double length = hand.getDigitLengths().get(i);
41              System.out.format("angle: %f , length: %f%n", angle, length);
42  
43              CvPoint fingertip = new CvPoint(
44                      (int) (centre.x() + length * Math.cos(angle)),
45                      (int) (centre.y() + length * Math.sin(angle))
46              );
47              cvDrawLine(copiedImage, centre, fingertip, CvScalar.RED, 10, CV_AA, 0);
48          }
49  
50          CvPoint dirPnt = new CvPoint(
51                  (int) (centre.x() + hand.getWidth() * Math.cos(hand.getPalm().getContourAxisAngle())),
52                  (int) (centre.y() + hand.getWidth() * Math.sin(hand.getPalm().getContourAxisAngle()))
53          );
54          cvDrawLine(copiedImage, centre, dirPnt, CvScalar.GREEN, 10, CV_AA, 0);
55  
56  
57          super.view(copiedImage);
58      }
