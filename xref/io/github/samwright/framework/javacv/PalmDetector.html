27      @Override
28      public Mediator process(Mediator input) {
29          Contour contour = (Contour) input.getData();
30  
31          // adapted from: http://www.javacodegeeks.com/2012/12/hand-and-finger-detection-using-javacv.html
32  
33          CvPoint cogPt = null;
34          int contourAxisAngle;
35          CvMoments moments = new CvMoments();
36          cvMoments(contour.getContour(), moments, 1);
37  
38          // center of gravity
39          double m00 = cvGetSpatialMoment(moments, 0, 0);
40          double m10 = cvGetSpatialMoment(moments, 1, 0);
41          double m01 = cvGetSpatialMoment(moments, 0, 1);
42  
43          if (m00 != 0) {   // calculate center
44              int xCenter = (int) Math.round(m10 / m00);
45              int yCenter = (int) Math.round(m01 / m00);
46              cogPt = new CvPoint(xCenter, yCenter);
47          }
48  
49          double m11 = cvGetCentralMoment(moments, 1, 1);
50          double m20 = cvGetCentralMoment(moments, 2, 0);
51          double m02 = cvGetCentralMoment(moments, 0, 2);
52          contourAxisAngle = calculateTilt(m11, m20, m02);
53  
54          return input.createNext(this, new Palm(cogPt, contourAxisAngle, contour.getSourceTaggedImage()));
55      }
56  
57      private int calculateTilt(double m11, double m20, double m02) {
58          double diff = m20 - m02;
59          if (diff == 0) {
60              if (m11 == 0)
61                  return 0;
62              else if (m11 > 0)
63                  return 45;
64              else   // m11 < 0
65                  return -45;
66          }
67  
68          double theta = 0.5 * Math.atan2(2 * m11, diff);
69          int tilt = (int) Math.round(Math.toDegrees(theta));
70  
71          if ((diff > 0) && (m11 == 0))
72              return 0;
73          else if ((diff < 0) && (m11 == 0))
74              return -90;
75          else if ((diff > 0) && (m11 > 0))  // 0 to 45 degrees
76              return tilt;
77          else if ((diff > 0) && (m11 < 0))  // -45 to 0
78              return (180 + tilt);   // change to counter-clockwise angle
79          else if ((diff < 0) && (m11 > 0))   // 45 to 90
80              return tilt;
81          else if ((diff < 0) && (m11 < 0))   // -90 to -45
82              return (180 + tilt);  // change to counter-clockwise angle
83  
84          return 0;
85      }
