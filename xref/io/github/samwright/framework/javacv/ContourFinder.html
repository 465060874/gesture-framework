44      @Override
45      public Mediator process(Mediator input) {
46          TaggedImage image = (TaggedImage) input.getData();
47          opencv_core.IplImage src = cvCloneImage(image.getImage());
48  
49          CvMemStorage storage = CvMemStorage.create();
50          CvSeq contours = new CvSeq(null);
51          cvFindContours(src, storage, contours, Loader.sizeof(opencv_core.CvContour.class),
52                  CV_RETR_EXTERNAL, CV_CHAIN_APPROX_NONE);
53  //        storage.release();
54  
55          // adapted from: http://www.javacodegeeks.com/2012/12/hand-and-finger-detection-using-javacv.html
56          // find the largest contour in the list based on bounded box size
57          float maxArea = 200;
58          CvBox2D maxBox = null;
59          CvSeq bigContour = null;
60  
61          while (contours != null && !contours.isNull()) {
62              if (contours.elem_size() > 0) {
63                  CvBox2D box = cvMinAreaRect2(contours, storage);
64                  if (box != null) {
65                      CvSize2D32f size = box.size();
66                      float area = size.width() * size.height();
67                      if (area > maxArea) {
68                          maxArea = area;
69                          bigContour = contours;
70                      }
71                  }
72              }
73              contours = contours.h_next();
74          }
75  
76          return input.createNext(this, new Contour(bigContour, image));
77      }
