59      public void setInputType(Class inputType) {
60          if (inputType.equals(this.inputType))
61              return;
62  
63          this.inputType = inputType;
64          viewerSelector.getItems().clear();
65  
66          for (Class clazz : ClassHelper.getAncestry(inputType)) {
67              DataViewer dataViewer = dataViewers.get(clazz);
68              if (dataViewer != null)
69                  viewerSelector.getItems().add(dataViewer.createClone().getComboItem());
70          }
71  
72          if (selectedViewer != null
73                  && selectedViewer.getViewableClass().isAssignableFrom(inputType)) {
74  
75              DataViewer newDataViewer = null;
76              for (DataViewer.ComboItem dataViewerComboItem : viewerSelector.getItems()) {
77                  DataViewer dataViewer = dataViewerComboItem.getDataViewer();
78                  if (dataViewer.getViewableClass().equals(selectedViewer.getViewableClass())) {
79                      newDataViewer = dataViewer;
80                      break;
81                  }
82              }
83              if (newDataViewer == null)
84                  viewerSelector.getSelectionModel().selectFirst();
85              else
86                  viewerSelector.getSelectionModel().select(newDataViewer.getComboItem());
87          } else {
88              viewerSelector.getSelectionModel().selectFirst();
89          }
90  
91          updateDataViewer();
92      }
93  
94      private void setDataViewer(DataViewer dataViewer) {
95          if (dataViewer == null)
96              dataViewer = new NullViewer();
97  
98          this.selectedViewer = dataViewer;
99          dataViewerBox.getChildren().clear();
100         dataViewerBox.getChildren().add(dataViewer);
101         updateDataViewer();
102     }

104     public void updateDataViewer() {
105         if (lastMediator == null)
106             return;
107 
108         setInputType(lastMediator.getData().getClass());
109 
110         if (selectedViewer != null)
111             selectedViewer.view(lastMediator);
112     }
113
