91      private void registerLinkWithElement(Element element) {
92          ToolboxController toolbox = (ToolboxController) MainApp.beanFactory.getBean("toolbox");
93  
94          if (element != null && !toolbox.getChildren().contains(this)) {
95              Set<ElementObserver> newObservers = new HashSet<>(element.getObservers());
96  
97              Iterator<ElementObserver> iterator = newObservers.iterator();
98              while (iterator.hasNext()) {
99                  ElementObserver observer = iterator.next();
100                 if (observer instanceof ElementLink)
101                     iterator.remove();
102             }
103 
104             if (elementLink != null)
105                 newObservers.add(elementLink);
106 
107             if (!newObservers.equals(element.getObservers())) {
108                 Element newModel = element.withObservers(newObservers);
109                 TopProcessor topProcessor = getModel().getTopProcessor();
110                 if (topProcessor != null)
111                     MainWindowController.getTopController().startTransientUpdateMode();
112                 try {
113                     element.replaceWith(newModel);
114                 } finally {
115                     if (topProcessor != null)
116                         MainWindowController.getTopController().endTransientUpdateMode();
117                 }
118             }
119         }
