14  /**
15   * A helper object that manages a list of children in an {@link EventuallyImmutable} parent-child
16   * hierarchy.
17   */
18  public class ChildrenManager<C extends ChildOf<P> & Processor,
19                               P extends ParentOf<C> & Processor>
20          implements ParentOf<C> {
21  
22      @Getter private List<C> children;
23      private final P managedParent;

25      /**
26       * Constructs a {@code ChildrenManager} with an empty children list.
27       *
28       * @param managedParent the parent who's children this will manage.
29       */
30      public ChildrenManager(P managedParent) {
31          this(managedParent, Collections.<C>emptyList());
32      }
33  
34      /**
35       * Constructs a {@code ChildrenManager} with the given children list.  The list won't be
36       * mutated (i.e. it is fine for children to be {@code Collections.unmodifiableList}).
37       *
38       * @param managedParent the parent who's children this will manage.
39       * @param children
40       */
41      public ChildrenManager(@NonNull P managedParent, @NonNull List<C> children) {
42          this.managedParent = managedParent;
43          this.children = children;
44      }
45  
46      @Override
47      @SuppressWarnings("unchecked")
48      public P withChildren(@NonNull List<C> newChildren) {
49          if (managedParent.isMutable()) {
50              if (new HashSet<>(newChildren).size() != newChildren.size())
51                  throw new RuntimeException("Supplied children list contains duplicates");
52  
53              children = Collections.unmodifiableList(new LinkedList<>(newChildren));
54              return managedParent;
55          } else {
56              P parentClone = (P) managedParent.createMutableClone();
57              return (P) parentClone.withChildren(newChildren);
58          }
59      }
60  
61      @Override
62      public boolean areChildrenValid() {
63          for (C child : children) {
64              if (!child.isValid())
65                  return false;
66              if (child instanceof ParentOf && !((ParentOf) child).areChildrenValid())
67                  return false;
68          }
69          return true;
70      }

72      /**
73       * Called before the managed parent is being fixed.
74       * <p/>
75       * One of four scenarios caused this update:
76       * <p/>
77       *     1. A child has mutated
78       *     <p/>
79       *     2. Another property of the this parent has been changed
80       *     <p/>
81       *     3. A child has been deleted or moved to another parent
82       *     <p/>
83       *     4. A new child has been added to this parent
84       *
85       * @param toReplace the parent that managedParent will replace.
86       */
87      @SuppressWarnings("unchecked")
88      public void beforeReplacing(P toReplace) {
89          if (!managedParent.isMutable())
90              return;
91  
92          List<C> latestChildren = new LinkedList<>();
93  
94          for (C child : children) {
95              C childNextVersion = (C) child.getNext();
96              if (childNextVersion == null) {
97                  if (child.getParent() == managedParent) {
98                      // Child is a new addition to this parent - just need to add it to list.
99                      latestChildren.add(child);
100                     if (!child.isReplacing())
101                         // Being loaded from XML - I need to fix it myself.
102                         child.replace(null);
103 
104                 } else {
105                     // Child has not been updated.  Grandfather it in to the new version:
106                     C newChild = (C) child.withParent(managedParent);
107                     child.replaceWith(newChild);
108                     latestChildren.add(newChild);
109                 }
110             } else {
111                 if (childNextVersion.getParent() == managedParent)
112                     // Child replaced itself from managedParent to managedParent.
113                     latestChildren.add(childNextVersion);
114                 // Otherwise, Child has been replaced and has been given to another parent,
115                 // so don't include it.
116             }
117         }
118 
119         children = Collections.unmodifiableList(latestChildren);
120     }

122     /**
123      * Called when the managed parent is having its replacement discarded,
124      * and discards all of its childrens' replacements.
125      */
126     public void discardNext() {
127         for (C child : children)
128             if (child.getNext() != null
129                     && !child.getNext().isMutable())
130                 child.discardNext();
131     }
132 
133     /**
134      * Called when the managed parent is having its older versions discarded,
135      * and discards all of its childrens' older versions.
136      */
137     public void discardPrevious() {
138         for (C child : children)
139             if (child.getPrevious() != null)
140                 child.discardPrevious();
141     }

143     /**
144      * Called when the managed parent is being set as the current version,
145      * and instructs its children to do the same.
146      */
147     public void setAsCurrentVersion() {
148         for (C child : children)
149             child.setAsCurrentVersion();
150     }

152     public void afterReplacement() {
153         for (C child : children)

158         Element childrenNode = doc.createElement("Children");

166     @SuppressWarnings("unchecked")
167     public void withXML(Element node, Map<UUID, Processor> dictionary) {
168         if (!managedParent.isMutable())
169             throw new RuntimeException("Should only be run when mutable");
170 
171         children = new ArrayList<>();
172         Element childrenNode = XMLHelper.getFirstChildWithName(node, "Children");
173 
174         for (Element childNode : XMLHelper.iterator(childrenNode)) {
175             C child = (C) ModelLoader.getPrototypeModel(childNode);
176             C newChild = (C) child.withXML(childNode, dictionary);
177 
178             newChild.withParent(managedParent);
179             newChild.setController(child.getController().createClone());
180             children.add(newChild);
181         }
