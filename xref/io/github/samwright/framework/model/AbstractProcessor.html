67          // Check typedata
68          if (completedTrainingBatch.getAll().isEmpty())
69              return completedTrainingBatch;
70  
71          Mediator first = completedTrainingBatch.getAll().iterator().next();
72          if (first.getData() != null) {
73              Processor creator = first.getHistory().getCreator();
74              if (creator != this) {
75                  throw new RuntimeException("Training batch was created by: " + creator
76                                          + " instead of this: " + this);
77              }
78  
79              Class<?> myOutput = getTypeData().getOutputType();
80              Class<?> outputDataType = first.getData().getClass();
81              if (!myOutput.isAssignableFrom(outputDataType))
82                  throw new RuntimeException("Training batch had output type: " + outputDataType
83                                           + " but this processor only outputs type: " + myOutput);
84          }
