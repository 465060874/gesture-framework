3   /**
4    * An object that can be replaced by another (or deleted), whilst keeping track of earlier and
5    * later versions.
6    */
7   public interface Replaceable {
8       /**
9        * Propose a replacement for this object.  How this is handled is determined by the concrete
10       * class implementing this.
11       * <p/>
12       * This must have no next version, and the replacement must have no previous version -
13       * otherwise a RuntimeException is thrown.
14       * <p/>
15       * As an example, a sequence of versions between {@code Replaceable}
16       * objects {@code start} and {@code end} can be contracted using
17       *
18       * <pre>{@code
19       * start.discardNext();  // start.getNext() == null
20       * end.discardPrevious(); // end.getPrevious() == null
21       * start.replaceWith(end);
22       * }</pre>
23       *
24       * @param replacement the replacement to propose.
25       * @throws RuntimeException if this object is still mutable,
26       *                          or has already been replaced or deleted.
27       */
28      void replaceWith(Replaceable replacement);

30      /**
31       * Propose that this object replace the supplied object.  If the given object is null,
32       * this becomes the start of a new line of objects.
33       *
34       * @param toReplace the object to replace.
35       */
36      void replace(Replaceable toReplace);
37  
38      /**
39       * This is called after all relevant objects have been fixed, and presents the implementing
40       * class with the opportunity to perform last-minute changes to itself now that all related
41       * objects have also been fixed.
42       */
43      void afterReplacement();
44  
45      /**
46       * Returns true iff this object is currently replacing another.
