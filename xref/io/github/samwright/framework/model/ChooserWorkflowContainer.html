11  /**
12   * A {@link WorkflowContainer} where one of its {@link Workflow} objects will be chosen for
13   * processing data.
14   * <p/>
15   * During the training phase this will process training data through all of its {@code Workflow}
16   * objects, but as the {@link CompletedTrainingBatch} flows back through it,
17   * it will decide on a strategy for choosing a {@code Workflow}.  The choice is made when
18   * {@code process(..)} is called, so the choosing strategy can involve information from the
19   * supplied {@link Mediator} (eg. its {@link History}).
20   */
21  public abstract class ChooserWorkflowContainer extends AbstractWorkflowContainer {

31      @Override
32      public Mediator process(Mediator input) {
33          Workflow chosenWorkflow = chooseWorkflow(input);
34          Mediator output = chosenWorkflow.process(input);
35  
36          ModelController controller = chosenWorkflow.getController();
37          if (controller != null)
38              controller.handleProcessedData(output);
39  
40          return output.createNext(this, output.getData());
41      }

43      public abstract Workflow chooseWorkflow(Mediator input);

45      @Override
46      public List<Mediator> processTrainingData(Mediator input) {
47          List<Mediator> outputs = new ArrayList<>(), finalOutputs = new ArrayList<>();
48  
49          for (Workflow workflow : getChildren()) {
50              List<Mediator> workflowOutput = workflow.processTrainingData(input);
51  
52              ModelController controller = workflow.getController();
53              if (controller != null)
54                  controller.handleProcessedTrainingData(workflowOutput);
55  
56              outputs.addAll(workflowOutput);
57          }
58  
59          for (Mediator output : outputs)
60              finalOutputs.add(output.createNext(this, output.getData()));
61  
62          return finalOutputs;
63      }

99              successfulOutputsByWorkflow.get(creator).add(successfulOutput);
100         }

102         for (Mediator output : batch.getAll()) {
103             Workflow creator = (Workflow) output.getHistory().getCreator();
104             allOutputsByWorkflow.get(creator).add(output);
105         }
106 
107         // Let workflows process their respective batches:

109         Map<Workflow, CompletedTrainingBatch> inputBatchesByWorkflow = new HashMap<>();
110         for (Workflow workflow : getChildren()) {

113                     successfulOutputsByWorkflow.get(workflow)

124         }

128                 = new HashMap<>();

131         // were this object's inputs).
132         for (Map.Entry<Workflow, CompletedTrainingBatch> e : inputBatchesByWorkflow.entrySet()) {
133             Workflow workflow = e.getKey();

135             Set<Mediator> successfulInputsForWorkflow = inputBatch.getSuccessful();
136 
137             // Make sure all input batches now share the same 'getAll()':
138             if (allInputs == null)
139                 allInputs = inputBatch.getAll();
140             else if (!allInputs.equals(inputBatch.getAll()))
141                 throw new RuntimeException("Child workflows did not properly rollback");
142 
143             // Index the input data that goes on to be successful by History and Workfow
144             for (Mediator successfulInputForWorkflow : successfulInputsForWorkflow) {
145                 History history = successfulInputForWorkflow.getHistory();
146 
147                 Map<Workflow, Set<Mediator>> successfulInputsForHistoryByWorkflow
148                         = successfulInputsByWorkflowAndHistory.get(history);
149 
150                 // If this is a new history object ...
151                 if (successfulInputsForHistoryByWorkflow == null) {
152                     // ... create a new map of workflows to successful mediators they processed ...
153                     successfulInputsForHistoryByWorkflow = new HashMap<>();
154                     // ... and assign an empty list for each workflow

157

159                 }

162                 // already been initialised as an empty HashSet).
163                 Set<Mediator> successfulInputsForWorkflowAndHistory
164                         = successfulInputsForHistoryByWorkflow.get(workflow);
165 
166                 // Finally, add the successful input to the set for this workflow and history.
167                 successfulInputsForWorkflowAndHistory.add(successfulInputForWorkflow);
168             }
169         }
170 
171         if (allInputs == null)
172             allInputs = Collections.emptySet();
173 
174         // Index allInputs by their History objects
175         Map<History,Set<Mediator>> allInputsByHistory = new HashMap<>();
176         for (Mediator mediator : allInputs) {
177             History history = mediator.getHistory();
178             Set<Mediator> allInputsForHistory = allInputsByHistory.get(history);
179 
180             if (allInputsForHistory == null) {
181                 allInputsForHistory = new HashSet<>();
182                 allInputsByHistory.put(history, allInputsForHistory);
183             }
184 
185             allInputsForHistory.add(mediator);
186         }
187 
188         // Let the concrete implementer use the completed training batch from the inputs of the
189         // workflows to decide how to choose the correct workflow in future.
190         handleSuccessfulInputsByWorkflowAndHistory(allInputsByHistory, successfulInputsByWorkflowAndHistory);
191 
192         // Find all input mediators that would have been chosen, if they now went to
193         // 'process(input)'
194         Set<Mediator> successfulChosenInputs = new HashSet<>();
195         Set<Mediator> allChosenInputs = new HashSet<>();
196         for (Workflow workflow : getChildren()) {
197             Set<Mediator> successfulWorkflowInputs
198                     = inputBatchesByWorkflow.get(workflow).getSuccessful();
199             Set<Mediator> allWorkflowInputs = inputBatchesByWorkflow.get(workflow).getAll();
200 
201             for (Mediator input : allWorkflowInputs)
202                 if (chooseWorkflow(input) == workflow)
203                     allChosenInputs.add(input);
204 
205             for (Mediator successfulInput : successfulWorkflowInputs)
206                 if (chooseWorkflow(successfulInput) == workflow)
207                     successfulChosenInputs.add(successfulInput);
208         }
209 
210         return new CompletedTrainingBatch(allChosenInputs, successfulChosenInputs);
