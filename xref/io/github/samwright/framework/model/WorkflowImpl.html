36      @Override
37      public boolean isValid() {
38          if (invalidlyOrderedElements == null) {
39              invalidlyOrderedElements = new LinkedList<>();
40  
41              if (getChildren().size() == 0) {
42                  if (!getTypeData().canBeEmptyContainer())
43                      invalidlyOrderedElements.add(null);
44  
45              } else {
46  
47                  Element firstChild = getChildren().get(0);
48                  if (!firstChild.getTypeData().canBeAtStartOfWorkflow(getTypeData()))
49                      invalidlyOrderedElements.add(null);
50  
51                  Element finalChild = getChildren().get(getChildren().size() - 1);
52                  if (!finalChild.getTypeData().canBeAtEndOfWorkflow(getTypeData()))
53                      invalidlyOrderedElements.add(finalChild);
54  
55                  Element previousElement = null;
56  
57                  for (Element element : getChildren()) {
58                      if (previousElement != null)
59                          if (!element.getTypeData().canComeAfter(previousElement.getTypeData()))
60                              invalidlyOrderedElements.add(previousElement);
61  
62                      previousElement = element;
63                  }
64              }
65          }
66  
67          return invalidlyOrderedElements.isEmpty();
68      }

70      @Override
71      public Mediator process(Mediator input) {
72          Mediator output = input;
73  
74          for (Element e : getChildren()) {
75              output = e.process(input);
76  
77              if (output == null)
78                  throw new NullPointerException("Element " + e + " returned null as processed data");
79  
80              for (ElementObserver observer : e.getObservers())
81                  observer.handleProcessedData(output);
82  
83              ModelController controller = e.getController();
84              if (controller != null)
85                  controller.handleProcessedData(output);
86  
87              input = output;
88          }
89  
90          return output.createNext(this, output.getData());
91      }

93      @Override
94      public List<Mediator> processTrainingData(Mediator firstInput) {
95          List<Mediator> outputs, inputs = Arrays.asList(firstInput);
96  
97          for (Element element : getChildren()) {
98              outputs = new LinkedList<>();
99              for (Mediator input : inputs)
100                 outputs.addAll(element.processTrainingData(input));
101 
102             if (outputs.isEmpty() && !inputs.isEmpty())
103                 throw new NullPointerException("Element " + element + " returned no training " +
104                         "data, even though it was given " + inputs.size());
105 
106             ModelController controller = element.getController();
107             if (controller != null)
108                 controller.handleProcessedTrainingData(outputs);
109 
110             for (ElementObserver observer : element.getObservers())
111                 observer.handleProcessedTrainingData(outputs);
112 
113             inputs = outputs;
114         }
115 
116         outputs = new ArrayList<>();
117         for (Mediator input : inputs)
118             outputs.add(input.createNext(this, input.getData()));
119         return outputs;
120     }

122     @Override
123     public CompletedTrainingBatch processCompletedTrainingBatch(CompletedTrainingBatch completedTrainingBatch) {
124         completedTrainingBatch = super.processCompletedTrainingBatch(completedTrainingBatch);
125         ListIterator<Element> itr = getChildren().listIterator(getChildren().size());
126 
127         while (itr.hasPrevious()) {
128             Element element = itr.previous();
129             completedTrainingBatch = element.processCompletedTrainingBatch(completedTrainingBatch);
130 
131             ModelController controller = element.getController();
132             if (controller != null)
133                 controller.handleTrained();
134         }
135 
136         return completedTrainingBatch;
137     }
