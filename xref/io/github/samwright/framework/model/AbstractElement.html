69      @Override
70      public void afterReplacement() {
71          if (this.dictionary == null) {
72              Set<ElementObserver> updatedObservers = new HashSet<>(observers);
73              for (ElementObserver observer : observers) {
74                  if (observer instanceof Processor)
75                      updatedObservers.add((ElementObserver) ((Processor) observer)
76                              .getCurrentVersion());
77                  else
78                      updatedObservers.add(observer);
79              }
80              observers = updatedObservers;
81          } else {
82              for (UUID uuid : observerUUIDs) {
83                  Processor observer = dictionary.get(uuid);
84                  if (observer == null)
85                      ModelLoader.getProcessor(uuid);
86                  if (observer != null)
87                      observers.add((ElementObserver) observer);
88              }
89              this.dictionary = null;
90          }
91      }

125         if (!isMutable())

127 
128         this.dictionary = map;
129         withParent(null);
130 
131         observerUUIDs = new HashSet<>();
132         org.w3c.dom.Element observersNode = XMLHelper.getFirstChildWithName(node, "Observers");
133 
134         for (org.w3c.dom.Element observerNode : XMLHelper.iterator(observersNode)) {
135             String uuidString = observerNode.getAttribute("UUID");
136             observerUUIDs.add(UUID.fromString(uuidString));
137         }
138 
139         TypeData newTypeData = typeData.withXML(node, map);
140 
141         return ((Element) super.withXML(node, map)).withTypeData(newTypeData);
